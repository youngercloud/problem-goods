常规的 DFS 题目，因为我们构造字符串的起点是 `t[0]`，所以需要在 `s` 中寻找 `s[i] == t[0]` 的字符作为 dfs 的起点

- 我们在任何时候都可以向左搜索，但是需要将 `turn` 改为 `1`
- 我们只有在 `turn == 0` 的时候才可以向右搜索

```
#pragma GCC optimize ("O3")
#pragma GCC optimize("unroll-loops")

#include <iostream> 
#include <sstream>

#include <map> 
#include <unordered_map>
#include <queue>

#include <algorithm> 
#include <functional> 

#define REP(i,n) for(int i=0;i<(n);i++)
#define all(cont) cont.begin(), cont.end()
#define EPS 1e-9

template<class T> void chmax(T & a, const T & b) { a = max(a, b); } 
template<class T> void chmin(T & a, const T & b) { a = min(a, b); } 
typedef long long LL;
LL mod = 1e9 + 7;
 
using namespace std;

// Code below
string s, t;
int solve = 0;
 
void dfs(int i, int j, int turn) {
    if (j == t.size() - 1) {
        solve = 1;
        return;
    }
 
    if (i < s.size() - 1 && s[i + 1] == t[j + 1] && turn == 0) {
        dfs(i + 1, j + 1, 0);
    }
 
    if (i > 0 && s[i - 1] == t[j + 1]) {
        dfs(i - 1, j + 1, 1);
    }
}
 
int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(nullptr);
 
    int q;
    cin >> q;
    while (q--) {
       solve = 0;
       cin >> s >> t;
       REP (i, s.size()) {
           if (s[i] == t[0]) {
               dfs(i, 0, 0);
           }
       }
       solve ? cout << "YES" << endl : cout << "NO" << endl;
    }
}
```
