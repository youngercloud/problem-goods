只需二分答案即可，我们需要满足的条件的是
- `mid <= a` 团队数不可超过或等于 `mathematicians`
- `mid <= b` 团队数不可超过或等于 `programmers`
- `4 * mid <= a + b` 由所有团队组成的总人数不可以超过两者的总数之和

void solve() {
    ll a, b;
    cin >> a >> b;
    ll left = 0;
    ll right = INT_MAX;
    while (left < right) {
        ll mid = left + (right - left + 1) / 2;
        if (mid <= a && mid <= b && 4 * mid <= a + b) {
            left = mid;
        } else {
            right = mid - 1;
        }
    }
    cout << left << endl;
}

    cin >> q;
    while (q--)
        solve();
    
}
